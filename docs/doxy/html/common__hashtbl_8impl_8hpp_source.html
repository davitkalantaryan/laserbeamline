161 static inline uint32_t hash1_( const void* a_pKey, uint32_t a_unKeySize )
162 {
163  register uint8_t *k = (uint8_t *)a_pKey;
164  register uint32_t a,b,c; /* the internal state */
165 
166  uint32_t len; /* how many key bytes still need mixing */
167 
168  /* Set up the internal state */
169  len = a_unKeySize;
170  a = b = 0x9e3779b9; /* the golden ratio; an arbitrary value */
171  c = 13; /* variable initialization of internal state */
172 
173  /*---------------------------------------- handle most of the key */
174  while (len >= 12)
175  {
176  a=a+(k[0]+((uint32_t)k[1]<<8)+((uint32_t)k[2]<<16) +((uint32_t)k[3]<<24));
177  b=b+(k[4]+((uint32_t)k[5]<<8)+((uint32_t)k[6]<<16) +((uint32_t)k[7]<<24));
178  c=c+(k[8]+((uint32_t)k[9]<<8)+((uint32_t)k[10]<<16)+((uint32_t)k[11]<<24));
179  mix2(a,b,c);
180  k = k+12; len = len-12;
181  }
182 
183 
184  /*------------------------------------- handle the last 11 bytes */
185  c = c+a_unKeySize;
186 
187  switch(len) /* all the case statements fall through */
188  {
189  case 11: c=c+((uint32_t)k[10]<<24);
190  case 10: c=c+((uint32_t)k[9]<<16);
191  case 9 : c=c+((uint32_t)k[8]<<8);
192 
193  /* the first byte of c is reserved for the length */
194  case 8 : b=b+((uint32_t)k[7]<<24);
195  case 7 : b=b+((uint32_t)k[6]<<16);
196  case 6 : b=b+((uint32_t)k[5]<<8);
197  case 5 : b=b+k[4];
198  case 4 : a=a+((uint32_t)k[3]<<24);
199  case 3 : a=a+((uint32_t)k[2]<<16);
200  case 2 : a=a+((uint32_t)k[1]<<8);
201  case 1 : a=a+k[0];
202  /* case 0: nothing left to add */
203  }
204  mix2(a,b,c);
205  /*-------------------------------------------- report the result */
206 
207  return c;
208 }
